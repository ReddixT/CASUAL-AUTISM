		context_type = player_context
        window_name = "template_ui_open_window"
		parent_window_token = top_bar
        visible = {
		}
		effects = {
			template_ui_open_button_click = {
				if = { 
                    limit = {
                        has_country_flag = opened_template_ui

                    clr_country_flag = opened_template_ui

                        set_country_flag = opened_template_ui
                    }
                }
			}
		}
	}

		context_type = player_context
        window_name = "template_ui_window"
		parent_window_token = top_bar

            has_country_flag = opened_template_ui
		}
		effects = {
			close_button_click = {
				clr_country_flag = opened_template_ui
			}

				20w_inf_effect = yes			
			}

			}

			}
